\newcommand{\grp}[5]{%
  \pgfmathsetmacro{\hb}{#2*\sc}\pgfmathsetmacro{\ha}{#3*\sc}%
  \pgfmathsetmacro{\bl}{#1-\bw-\ing/2}\pgfmathsetmacro{\br}{#1-\ing/2}%
  \pgfmathsetmacro{\al}{#1+\ing/2}\pgfmathsetmacro{\ar}{#1+\bw+\ing/2}%
  \pgfmathsetmacro{\cb}{(\bl+\br)/2}\pgfmathsetmacro{\ca}{(\al+\ar)/2}%
  \fill[slateA!80] (\bl,0) rectangle (\br,\hb);
  \fill[purpleBrand] (\al,0) rectangle (\ar,\ha);
  \node[white,font=\Large\bfseries,anchor=north] at (\cb,{\hb-0.12}) {#2};
  \node[white,font=\Large\bfseries,anchor=north] at (\ca,{\ha-0.12}) {#3};
  \node[passThree,font=\huge\bfseries,anchor=south] at (\ca,{\ha+0.14}) {+#4};
  \node[inkA,font=\Large,anchor=north] at (#1,-0.2) {#5};
}
